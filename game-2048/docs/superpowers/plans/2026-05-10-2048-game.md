```javascript
const SIZE = 4;
let board = [];
let score = 0;
let best = 0;

function initBoard() {
    board = Array.from({length: SIZE}, () => Array(SIZE).fill(0));
    score = 0;
    addRandomTile();
    addRandomTile();
    updateScore();
}

function getEmptyCells() {
    const empty = [];
    for (let r = 0; r < SIZE; r++) {
        for (let c = 0; c < SIZE; c++) {
            if (board[r][c] === 0) empty.push({r, c});
        }
    }
    return empty;
}

function addRandomTile() {
    const empty = getEmptyCells();
    if (empty.length === 0) return;
    const {r, c} = empty[Math.floor(Math.random() * empty.length)];
    board[r][c] = Math.random() < 0.9 ? 2 : 4;
}

function updateScore() {
    document.getElementById('score').textContent = score;
    if (score > best) {
        best = score;
        document.getElementById('best').textContent = best;
    }
}

function updateBest() {
    document.getElementById('best').textContent = best;
}
```

- [ ] **Step 2: Add move and merge logic**

```javascript
function slideAndMerge(row) {
    let arr = row.filter(v => v !== 0);
    for (let i = 0; i < arr.length - 1; i++) {
        if (arr[i] === arr[i + 1]) {
            arr[i] *= 2;
            score += arr[i];
            arr[i + 1] = 0;
        }
    }
    arr = arr.filter(v => v !== 0);
    while (arr.length < SIZE) arr.push(0);
    return arr;
}

function moveBoard(direction) {
    const oldBoard = board.map(row => [...row]);
    const oldScore = score;

    if (direction === 'left') {
        for (let r = 0; r < SIZE; r++) {
            board[r] = slideAndMerge(board[r]);
        }
    } else if (direction === 'right') {
        for (let r = 0; r < SIZE; r++) {
            board[r] = slideAndMerge([...board[r]].reverse()).reverse();
        }
    } else if (direction === 'up') {
        for (let c = 0; c < SIZE; c++) {
            let col = [board[0][c], board[1][c], board[2][c], board[3][c]];
            col = slideAndMerge(col);
            for (let r = 0; r < SIZE; r++) board[r][c] = col[r];
        }
    } else if (direction === 'down') {
        for (let c = 0; c < SIZE; c++) {
            let col = [board[3][c], board[2][c], board[1][c], board[0][c]];
            col = slideAndMerge(col);
            for (let r = 0; r < SIZE; r++) board[3 - r][c] = col[r];
        }
    }

    const changed = !board.every((row, r) =>
        row.every((v, c) => v === oldBoard[r][c])
    );

    if (changed) {
        addRandomTile();
        updateScore();
    }

    return {changed, oldBoard};
}
```

- [ ] **Step 3: Add win/lose detection**

```javascript
function hasWon() {
    return board.some(row => row.some(v => v === 2048));
}

function hasLost() {
    if (getEmptyCells().length > 0) return false;
    for (let r = 0; r < SIZE; r++) {
        for (let c = 0; c < SIZE; c++) {
            if (c < SIZE - 1 && board[r][c] === board[r][c + 1]) return false;
            if (r < SIZE - 1 && board[r][c] === board[r + 1][c]) return false;
        }
    }
    return true;
}
```

```javascript
const tileContainer = document.getElementById('tile-container');
const CELL_SIZE = 108; // (480 - 24 - 36) / 4 = 105, let's use 105 + 12 gap

function getTileColor(value) {
    const colors = {
        2:    {bg: '#eee4da', fg: '#776e65'},
        4:    {bg: '#ede0c8', fg: '#776e65'},
        8:    {bg: '#f2b179', fg: '#f9f6f2'},
        16:   {bg: '#f59563', fg: '#f9f6f2'},
        32:   {bg: '#f67c5f', fg: '#f9f6f2'},
        64:   {bg: '#f65e3b', fg: '#f9f6f2'},
        128:  {bg: '#edcf72', fg: '#f9f6f2'},
        256:  {bg: '#edcc61', fg: '#f9f6f2'},
        512:  {bg: '#edc850', fg: '#f9f6f2'},
        1024: {bg: '#edc53f', fg: '#f9f6f2'},
        2048: {bg: '#edc22e', fg: '#f9f6f2'},
    };
    return colors[value] || {bg: '#3c3a32', fg: '#f9f6f2'};
}

function renderBoard() {
    tileContainer.innerHTML = '';
    for (let r = 0; r < SIZE; r++) {
        for (let c = 0; c < SIZE; c++) {
            if (board[r][c] === 0) continue;
            const tile = document.createElement('div');
            tile.className = 'tile';
            tile.textContent = board[r][c];
            const {bg, fg} = getTileColor(board[r][c]);
            tile.style.backgroundColor = bg;
            tile.style.color = fg;
            tile.style.width = tile.style.height = CELL_SIZE + 'px';
            tile.style.transform = `translate(${c * (CELL_SIZE + GAP)}px, ${r * (CELL_SIZE + GAP)}px)`;
            const fontSize = board[r][c] >= 1000 ? '32px' : board[r][c] >= 100 ? '36px' : '40px';
            tile.style.fontSize = fontSize;
            tileContainer.appendChild(tile);
        }
    }
}
```

- [ ] **Step 1: Add keyboard handler and message display**

```javascript
let processing = false;

function showMessage(text) {
    const msg = document.getElementById('message');
    msg.textContent = text;
    msg.classList.remove('hidden');
}

function hideMessage() {
    document.getElementById('message').classList.add('hidden');
}

function handleMove(direction) {
    if (processing) return;
    processing = true;

    const {changed, oldBoard} = moveBoard(direction);

    if (changed) {
        renderBoard();
        if (hasWon()) showMessage('You Win!');
        else if (hasLost()) showMessage('Game Over');
    }

    setTimeout(() => { processing = false; }, 150);
}

document.addEventListener('keydown', (e) => {
    const keys = {
        'ArrowLeft': 'left',
        'ArrowRight': 'right',
        'ArrowUp': 'up',
        'ArrowDown': 'down',
    };
    if (keys[e.key]) {
        e.preventDefault();
        handleMove(keys[e.key]);
    }
});
```
